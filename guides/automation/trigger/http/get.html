Automation with HTTP GET trigger

index.ts
import { App, type AppSchema } from '@latechforce/engine';

const schema: AppSchema = {
  automations: [
    {
      id: 1,
      name: 'get',
      trigger: {
        service: 'http',
        event: 'get',
        params: {
          path: 'get',
        },
      },
      actions: [],
    },
  ],
};

await new App().start(schema);
